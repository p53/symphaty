sub openURL { 

	my $self = shift;
	my $url = shift;
	my $browsers = $self->{browsers};
	
	if($self->{OS} eq 'MSWin32') {
		system('start /b ' . $url);
	} elsif($self->{OS} eq 'linux' or $self->{OS} eq 'darwin') {
		
		foreach my $browser(@$browsers) {
			my $isThere = `whereis $browser`;
			if($isThere =~ /^[\w-]+\:(\s+[\w\/\-_]+)/) {
				my $cmd = `$1 $url`;
				last;
			} # if
		} # foreach
		
	} # if

} # end sub openURL
